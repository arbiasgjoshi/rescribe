import { List, Tab, TabGroup } from '@i4o/rescribe'

# Tabs

Toggle between different content with a tab interface.

<TabGroup>
    <Tab title='Tab 1'>
        Content of tab 1.
    </Tab>
    <Tab title='Tab 2'>
        Content of tab 2.
    </Tab>
    <Tab title='Tab 3'>
        Content of tab 3.
    </Tab>
</TabGroup>

## Props

### Tab

<List
    items={[
        {
            title: 'title',
            description: 'Title of the tab',
            type: 'string | ReactNode',
            required: true
        },
    ]}
/>
